{
    Кодер:
    {
        Функции для работы с ОТ:
        {
            intpitch - определение целочисленного значения ОТ
            double_ver - удаление низкого ОТ
            fpr - Улучшение дробного значения ОТ
            pitch3 - Окончательное определение ОТ
            pitch2 - Улучшение дробного значения ОТ в полосе 0-500 Гц
            double_ck - Определение ОТ с удвоенной точностью
        }
        Фильтры:
        {
            d_lpc - синтезирующий LPC фильтр
            melp_APU - Медианный фильтр
            melp_lpc - LPC Анализ
        }
        Квантовые усиления:
        {
            melp_Qgain - квантование усиления
            mekp_Qpitch - квантование усиления
        }
        ЛСЧ, определения и конверторы:
        {
            d_lsf - определение вектора квантованных ЛСЧ
            lpc_residual - Определение остатка предсказания
            lsf_clmp - Регулирует расстояние между граничными ЛСЧ
            melp_5b - Расчет 5 полосовых сигналов и огибающих для 2-5 полос
            melp_bpva - Вариант определения вокализованности полос
            melp_FMCQ - Векторный квантователь амплитуд Фурье - спектра
            melp_lpc2lsf - Конвертирует коэффициенты ЛП в ЛСЧ
            melp_lsf2lpc - Конвертер LSF в LPC
            melp_msvq - квантование коэффициентов линейного предсказания
        }
    }
    Декодер:
    {
        Оценка и подавление шумов:
        {
            noise_est - Оценка шума
            noise_sup - подавление шума
        }
        Фильтры:
        {
            melp_lsf2lpc - конвертер LSF в LPC
            d_disp - Импульсный дисперсионный фильтр
        }
        Коррекция, регулировка и восстановление:
        {
            d_ga - регулирование масштаба выходного речевого сигнала
            d_gains - восстановление усилений
            d_ase - адаптивная спектральная коррекция
        }
        Расчет, определение и формирование:
        {
            d_mix(d_min) - Формирование смешанного сигнала возбуждения
            d_lsf - Определение вектора квантованных ЛСЧ
            d_k1 - Расчет первого коэффициента отражения
        }
    }
}
